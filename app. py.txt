from flask import Flask, request, jsonify

app = Flask(__name__)

GEMINI_API_KEY = "YOUR_KEY_HERE"   # Lwin will add manually

def auto_smart_reply(text, mode="normal"):
    if mode == "director":
        return f"Done. လုပ်ပြီးပါပြီ Lwin."
    if mode == "system":
        return f"System updated. စနစ်ပြင်ဆင်ပြီးပါပြီ."
    if mode == "alert":
        return f"သတိပေးချက်ရောက်ပြီ Lwin. Alert received."
    if mode == "ai_missing":
        return f"AI key missing. Gemini key ထည့်ပြီးမှ AI reply ပေးနိုင်မယ်."
    return f"အဖြေက ဒီလိုပါ… Here is your answer…"

def detect_mode(msg):
    msg = msg.lower()
    if "lwin" in msg or msg.startswith("/director"):
        return "director"
    if msg.startswith("/sys"):
        return "system"
    if msg.startswith("/alert"):
        return "alert"
    if msg.startswith("/ai"):
        return "ai"
    return "normal"

@app.post("/bot")
def bot():
    data = request.json
    msg = data.get("message", "")
    mode = detect_mode(msg)

    if mode == "ai":
        if GEMINI_API_KEY == "YOUR_KEY_HERE":
            reply = auto_smart_reply(msg, "ai_missing")
        else:
            reply = "AI processing will run here."
    else:
        reply = auto_smart_reply(msg, mode)

    return jsonify({"reply": reply, "status": "ok"})

@app.get("/")
def home():
    return "Universal-authority server running."
